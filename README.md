function progressBar(progress, total, width = 20) {
    const percent = Math.min(100, Math.floor((progress / total) * 100));
    const filled = Math.floor((width * progress) / total);
    const bar = '‚ñà'.repeat(filled) + '‚ñë'.repeat(width - filled);
    const timeLeft = Math.ceil((total - progress) / 60);
    return `[${bar}] ${progress}/${total} (${percent}%) ‚Ä¢ ${timeLeft} min left`;
}

    // 8. –û—Å–Ω–æ–≤–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ –∑–∞–¥–∞–Ω–∏–π (—Ä–µ–∫—É—Ä—Å–∏—è)

                                removed: [fakeGame],
                                added: [],
                                games: []
                            });

                            FluxDispatcher.unsubscribe("QUESTS_SEND_HEARTBEAT_SUCCESS", onHeartbeat);
                            processNextQuest();
                        }
                    };

                    FluxDispatcher.subscribe("QUESTS_SEND_HEARTBEAT_SUCCESS", onHeartbeat);

                    info(`–í–∞—à–∞ –∏–≥—Ä–∞ –±—ã–ª–∞ –ø–æ–¥–º–µ–Ω–µ–Ω–∞ –ø–æ–¥ ${applicationName}. –ü–æ–¥–æ–∂–¥–∏—Ç–µ –µ—â–µ ${Math.ceil((secondsNeeded - secondsDone) / 60)} –º–∏–Ω—É—Ç.`);

            if (!isDesktopApp) {
